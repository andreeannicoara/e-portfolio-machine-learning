from scipy.optimize import linear_sum_assignment

def best_cluster_label_mapping(y_true, cluster_labels):
    """
    Returns:
      mapping dict: cluster_id -> true_label
      y_pred_mapped: mapped predictions aligned to true labels
    """
    y_true = np.asarray(y_true)
    cluster_labels = np.asarray(cluster_labels)

    true_labels = np.unique(y_true)
    clusters = np.unique(cluster_labels)

    cost = np.zeros((len(clusters), len(true_labels)), dtype=int)
    for i, c in enumerate(clusters):
        for j, t in enumerate(true_labels):
            cost[i, j] = np.sum((cluster_labels == c) & (y_true == t))

    row_ind, col_ind = linear_sum_assignment(cost.max() - cost)

    mapping = {clusters[r]: true_labels[c] for r, c in zip(row_ind, col_ind)}
    y_pred_mapped = np.array([mapping[c] for c in cluster_labels])

    return mapping, y_pred_mapped

# Load
df = pd.read_csv("Unit06 iris.csv")

# Identify label column
# Common names: "species" or last column
if "species" in df.columns:
    y = df["species"]
    X = df.drop(columns=["species"])
else:
    y = df.iloc[:, -1]
    X = df.iloc[:, :-1]

# Keep numeric only, drop anything non-numeric just in case
X = X.select_dtypes(include=[np.number])

# Pipeline: impute, scale, KMeans
pipe = Pipeline([
    ("imputer", SimpleImputer(strategy="median")),
    ("scaler", StandardScaler()),
    ("kmeans", KMeans(n_clusters=3, random_state=42, n_init=10))
])

cluster_labels = pipe.fit_predict(X)

mapping, y_pred = best_cluster_label_mapping(y, cluster_labels)

print("Cluster to label mapping:", mapping)
print("Accuracy after best matching:", accuracy_score(y, y_pred))

# Confusion style table
ct = pd.crosstab(pd.Series(y, name="True"), pd.Series(y_pred, name="Pred"))
print(ct)

import time 
from sklearn.preprocessing import OneHotEncoder
from sklearn.decomposition import TruncatedSVD

t0 = time.time()

def mark(msg):
    print(f"{msg} | {time.time()-t0:.1f}s")

df = pd.read_csv("Unit06 weatherAUS.csv")
df = df.sample(n=20000, random_state=42)
mark("Loaded and sampled")

drop_cols = [c for c in ["Date", "RainTomorrow"] if c in df.columns]
X = df.drop(columns=drop_cols)

num_cols = X.select_dtypes(include=[np.number]).columns.tolist()
cat_cols = [c for c in X.columns if c not in num_cols]

numeric_transformer = Pipeline([
    ("imputer", SimpleImputer(strategy="median")),
    ("scaler", StandardScaler()),
])

categorical_transformer = Pipeline([
    ("imputer", SimpleImputer(strategy="most_frequent")),
    ("onehot", OneHotEncoder(handle_unknown="ignore", sparse_output=True)),
])

preprocess = ColumnTransformer([
    ("num", numeric_transformer, num_cols),
    ("cat", categorical_transformer, cat_cols),
], remainder="drop")

X_proc = preprocess.fit_transform(X)
mark(f"Preprocessed, shape={X_proc.shape}")

# PCA like 2D projection for sparse matrices
svd = TruncatedSVD(n_components=2, random_state=42)
X2 = svd.fit_transform(X_proc)
mark("2D projection done")

results = []
rng = np.random.RandomState(42)
plot_idx = rng.choice(len(X2), size=min(5000, len(X2)), replace=False)

for k in range(2, 7):
    print("Running K =", k)
    km = KMeans(n_clusters=k, random_state=42, n_init=10)
    labels = km.fit_predict(X_proc)
    mark(f"K={k} fitted")

    # Silhouette on a subset only, otherwise it is very slow
    sil_idx = rng.choice(X_proc.shape[0], size=min(2000, X_proc.shape[0]), replace=False)
    sil = silhouette_score(X_proc[sil_idx], labels[sil_idx])

    results.append({"K": k, "inertia": km.inertia_, "silhouette_subsample": sil})

    plt.figure()
    plt.scatter(X2[plot_idx, 0], X2[plot_idx, 1], c=np.array(labels)[plot_idx])
    plt.title(f"WeatherAUS clusters, 2D projection (K={k})")
    plt.xlabel("Comp1")
    plt.ylabel("Comp2")
    plt.show()

pd.DataFrame(results)
